set(target_libraries
    gridpack_components
    gridpack_parallel
    ${Boost_LIBRARIES}
    ${GA_LIBRARIES}
    ${MPI_CXX_LIBRARIES})

include_directories(BEFORE ${CMAKE_CURRENT_SOURCE_DIR})

add_library(gridpack_ymatrix_components
  ymatrix_components.cpp
  )

target_link_libraries(gridpack_ymatrix_components gridpack_components)

gridpack_set_library_version(gridpack_ymatrix_components)

# -------------------------------------------------------------
# shared library dependencies
# -------------------------------------------------------------
if (GRIDPACK_LIB_LINK_LIBRARIES)
  target_link_libraries(gridpack_ymatrix_components
    gridpack_components
    )
endif()

# -------------------------------------------------------------
# installation
# -------------------------------------------------------------
install(FILES 
  ymatrix_components.hpp  
  DESTINATION include/gridpack/applications/components/y_matrix
)

install(TARGETS 
  gridpack_ymatrix_components
  DESTINATION lib
)
